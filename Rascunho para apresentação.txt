	package controller;

import model.Aluno;
import model.Plano;
import utils.ArquivoUtil;

import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;

public class AlunoController {

    private static final String ARQUIVO = "dados/alunos.txt";

    public List<Aluno> listarAlunos() {
        List<Aluno> alunos = new ArrayList<>();
        List<String> linhas = ArquivoUtil.lerLinhas(ARQUIVO);

        for (String linha : linhas) {
            alunos.add(Aluno.fromLinhaArquivo(linha));
        }

        return alunos;
    }

    public void salvarAlunos(List<Aluno> alunos) {
        List<String> linhas = new ArrayList<>();
        for (Aluno aluno : alunos) {
            linhas.add(aluno.paraLinhaArquivo());
        }
        ArquivoUtil.escreverLinhas(ARQUIVO, linhas);
    }

    public int gerarProximoId() {
        List<Aluno> alunos = listarAlunos();
        int maiorId = 0;
        for (Aluno aluno : alunos) {
            if (aluno.getId() > maiorId) {
                maiorId = aluno.getId();
            }
        }
        return maiorId + 1;
    }

    public void cadastrarAluno(String nome, String cpf, int idade, String nomePlano, LocalDate dataInicio) {
        int id = gerarProximoId();
        Plano plano = Plano.criarPorNome(nomePlano);
        LocalDate dataVencimento = dataInicio.plusDays(30);

        Aluno aluno = new Aluno(id, nome, cpf, idade, plano, dataInicio, dataVencimento);

        List<Aluno> alunos = listarAlunos();
        alunos.add(aluno);
        salvarAlunos(alunos);
    }

    public Aluno buscarPorId(int id) {
        List<Aluno> alunos = listarAlunos();
        for (Aluno aluno : alunos) {
            if (aluno.getId() == id) {
                return aluno;
            }
        }
        return null;
    }

    public boolean removerAluno(int id) {
        List<Aluno> alunos = listarAlunos();
        Aluno alunoEncontrado = null;

        for (Aluno aluno : alunos) {
            if (aluno.getId() == id) {
                alunoEncontrado = aluno;
                break;
            }
        }

        if (alunoEncontrado == null) {
            return false;
        }

        alunos.remove(alunoEncontrado);
        salvarAlunos(alunos);
        return true;
    }

    public boolean atualizarPlano(int id, String nomePlano) {
        List<Aluno> alunos = listarAlunos();

        for (Aluno aluno : alunos) {
            if (aluno.getId() == id) {
                Plano novoPlano = Plano.criarPorNome(nomePlano);
                aluno.setPlano(novoPlano);
                salvarAlunos(alunos);
                return true;
            }
        }

        return false;
    }

    public boolean renovarPlano(int id) {
        List<Aluno> alunos = listarAlunos();

        for (Aluno aluno : alunos) {
            if (aluno.getId() == id) {
                LocalDate novaData = LocalDate.now();
                aluno.setDataInicio(novaData);
                aluno.setDataVencimento(novaData.plusDays(30));
                salvarAlunos(alunos);
                return true;
            }
        }

        return false;
    }
    public Aluno buscarPorCpf(String cpf) {

    List<Aluno> alunos = listarAlunos();

    for (Aluno aluno : alunos) {
        if (aluno.getCpf().equals(cpf)) {
            return aluno;
        }
    }
    return null;
}
}

	package controller;

import model.RegistroAcesso;
import utils.ArquivoUtil;

import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;

public class RegistroAcessoController {

    private static final String ARQUIVO = "dados/registros.txt";

    public List<RegistroAcesso> listarRegistros() {
        List<RegistroAcesso> registros = new ArrayList<>();
        List<String> linhas = ArquivoUtil.lerLinhas(ARQUIVO);

        for (String linha : linhas) {
            registros.add(RegistroAcesso.fromLinhaArquivo(linha));
        }

        return registros;
    }

    public void salvarRegistros(List<RegistroAcesso> registros) {
        List<String> linhas = new ArrayList<>();
        for (RegistroAcesso registro : registros) {
            linhas.add(registro.paraLinhaArquivo());
        }
        ArquivoUtil.escreverLinhas(ARQUIVO, linhas);
    }

    public void registrarEntrada(int idAluno, String nomeAluno) {
        RegistroAcesso registro = new RegistroAcesso(idAluno, nomeAluno, LocalDateTime.now(), null);
        List<RegistroAcesso> registros = listarRegistros();
        registros.add(registro);
        salvarRegistros(registros);
    }

    public boolean registrarSaida(int idAluno) {
        List<RegistroAcesso> registros = listarRegistros();

        for (int i = registros.size() - 1; i >= 0; i--) {
            RegistroAcesso registro = registros.get(i);
            if (registro.getIdAluno() == idAluno && registro.getDataHoraSaida() == null) {
                registro.setDataHoraSaida(LocalDateTime.now());
                salvarRegistros(registros);
                return true;
            }
        }

        return false;
    }

    public boolean alunoEstaDentro(int idAluno) {
        List<RegistroAcesso> registros = listarRegistros();

        for (int i = registros.size() - 1; i >= 0; i--) {
            RegistroAcesso registro = registros.get(i);
            if (registro.getIdAluno() == idAluno) {
                return registro.getDataHoraSaida() == null;
            }
        }

        return false;
    }
}

public class RelatorioController {

    private AlunoController alunoController;
    private RegistroAcessoController registroController;

    public RelatorioController() {
        this.alunoController = new AlunoController();
        this.registroController = new RegistroAcessoController();
    }

    public List<Aluno> listarAlunosAtivos() {
        List<Aluno> ativos = new ArrayList<>();
        for (Aluno aluno : alunoController.listarAlunos()) {
            if (aluno.getStatus().equals("Ativo")) {
                ativos.add(aluno);
            }
        }
        return ativos;
    }

    public List<Aluno> listarAlunosVencidos() {
        List<Aluno> vencidos = new ArrayList<>();
        for (Aluno aluno : alunoController.listarAlunos()) {
            if (aluno.getStatus().equals("Vencido")) {
                vencidos.add(aluno);
            }
        }
        return vencidos;
    }

    public int contarPorPlano(String nomePlano) {
        int total = 0;
        for (Aluno aluno : alunoController.listarAlunos()) {
            if (aluno.getPlano().getNome().equalsIgnoreCase(nomePlano)) {
                total++;
            }
        }
        return total;
    }

    public List<RegistroAcesso> historicoAcessos() {
        return registroController.listarRegistros();
    }
}

public class Aluno extends Pessoa {

    private String personal;

    private static final DateTimeFormatter FORMATO_DATA = DateTimeFormatter.ofPattern("dd/MM/yyyy");

    public Aluno(int id, String nome, String cpf, int idade, Plano plano, LocalDate dataInicio, LocalDate dataVencimento) {
        super(id, nome, cpf, idade);
        this.plano = plano;
        this.dataInicio = dataInicio;
        this.dataVencimento = dataVencimento;
        this.personal = "Nenhum";
    }

    }

    public void setPlano(Plano plano) {
        this.plano = plano;
    }

    }

        this.dataInicio = dataInicio;
    }

        return dataVencimento;
    }
    
    public String getPersonal() {
    return personal;

    }

    public void setPersonal(String personal) {
    this.personal = personal;
}


    public void setDataVencimento(LocalDate dataVencimento) {
        this.dataVencimento = dataVencimento;
    }

    public String getStatus() {
        LocalDate hoje = LocalDate.now();
        if (hoje.isAfter(dataVencimento)) {
            return "Vencido";
        }
        return "Ativo";
    }

    public String toString() {
        return "ID: " + id
                + " | Nome: " + nome
                + " | CPF: " + cpf
                + " | Idade: " + idade
                + " | Plano: " + plano.getNome()
                + " | Inicio: " + dataInicio.format(FORMATO_DATA)
                + " | Vencimento: " + dataVencimento.format(FORMATO_DATA)
                + " | Status: " + getStatus()
                + " | Personal: " + personal;
    }

    @Override
    public String paraLinhaArquivo() {
        return id + ";" + nome + ";" + cpf + ";" + idade + ";" + plano.getNome() + ";"
                + dataInicio.format(FORMATO_DATA) + ";" + dataVencimento.format(FORMATO_DATA);
    }

    public static Aluno fromLinhaArquivo(String linha) {
        String[] partes = linha.split(";");
        int id = Integer.parseInt(partes[0]);
        String nome = partes[1];
        String cpf = partes[2];
        int idade = Integer.parseInt(partes[3]);
        Plano plano = Plano.criarPorNome(partes[4]);
        LocalDate inicio = LocalDate.parse(partes[5], FORMATO_DATA);
        LocalDate vencimento = LocalDate.parse(partes[6], FORMATO_DATA);
        return new Aluno(id, nome, cpf, idade, plano, inicio, vencimento);
    }

	package utils;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

public class ArquivoUtil {

    public static List<String> lerLinhas(String caminho) {
        List<String> linhas = new ArrayList<>();
        File arquivo = new File(caminho);

        if (!arquivo.exists()) {
            return linhas;
        }

        try (BufferedReader leitor = new BufferedReader(new FileReader(arquivo))) {
            String linha;
            while ((linha = leitor.readLine()) != null) {
                if (!linha.trim().isEmpty()) {
                    linhas.add(linha);
                }
            }
        } catch (IOException e) {
            System.out.println("Erro ao ler arquivo: " + e.getMessage());
        }

        return linhas;
    }

    public static void escreverLinhas(String caminho, List<String> linhas) {
        try (BufferedWriter escritor = new BufferedWriter(new FileWriter(caminho))) {
            for (String linha : linhas) {
                escritor.write(linha);
                escritor.newLine();
            }
        } catch (IOException e) {
            System.out.println("Erro ao escrever arquivo: " + e.getMessage());
        }
    }

    public static void adicionarLinha(String caminho, String linha) {
        try (BufferedWriter escritor = new BufferedWriter(new FileWriter(caminho, true))) {
            escritor.write(linha);
            escritor.newLine();
        } catch (IOException e) {
            System.out.println("Erro ao escrever arquivo: " + e.getMessage());
        }
    }
}

    }

    public static boolean numeroInteiroValido(String texto) {
        try {
            Integer.parseInt(texto);
            return true;
        } catch (NumberFormatException e) {
            return false;
        }
    }

    public static boolean dataValida(String data) {
        try {
            LocalDate.parse(data, FORMATO_DATA);
            return true;
        } catch (DateTimeParseException e) {
            return false;
        }
    }

    public static LocalDate converterParaData(String data) {
        return LocalDate.parse(data, FORMATO_DATA);
    }

    public static boolean planoValido(String plano) {
        String p = plano.toUpperCase();
        return p.equals("BASICO") || p.equals("PREMIUM") || p.equals("VIP");

import controller.LoginController;

public class LoginView {

    private Scanner scanner;
    private LoginController loginController;

    public LoginView(Scanner scanner) {
        this.scanner = scanner;
        this.loginController = new LoginController();
    }

    public boolean iniciar() {
        if (!loginController.existeUsuarioCadastrado()) {
            cadastrarPrimeiroUsuario();
        }

        return realizarLogin();
    }

    private void cadastrarPrimeiroUsuario() {
        System.out.println("=== CADASTRO DO PRIMEIRO USUARIO ===");
        System.out.println("Nenhum usuario cadastrado. Realize o cadastro inicial.");

        System.out.print("Digite o login: ");
        String login = scanner.nextLine();

        System.out.print("Digite a senha: ");
        String senha = scanner.nextLine();

        loginController.cadastrarUsuario(login, senha);
        System.out.println("Usuario cadastrado com sucesso!");
    }

    private boolean realizarLogin() {
        int tentativas = 0;

        while (tentativas < 3) {
            System.out.println("=== LOGIN ===");
            System.out.print("Login: ");
            String login = scanner.nextLine();

            System.out.print("Senha: ");
            String senha = scanner.nextLine();

            if (loginController.validarLogin(login, senha)) {
                System.out.println("Login realizado com sucesso!");
                return true;
            }

            tentativas++;
            System.out.println("Login ou senha incorretos. Tentativas restantes: " + (3 - tentativas));
        }

        System.out.println("Voce excedeu o numero de tentativas.");
        return solicitarNovaSenha();
    }

    private boolean solicitarNovaSenha() {
        System.out.println("=== REDEFINICAO DE SENHA ===");

        while (true) {
            System.out.print("Digite uma nova senha: ");
            String novaSenha = scanner.nextLine();

            if (loginController.senhaUsadaAnteriormente(novaSenha)) {
                System.out.println("Essa senha ja foi utilizada anteriormente. Escolha outra.");
                continue;
            }

            loginController.alterarSenha(novaSenha);
            System.out.println("Senha alterada com sucesso! Faca login novamente.");
            return realizarLogin();
        }
    }
